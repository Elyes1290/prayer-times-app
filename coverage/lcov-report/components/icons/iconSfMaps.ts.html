/** Correspondances Material Icons → SF Symbol. */
export const MI_TO_SF = {
  "chevron-left": "chevron.left",
  "chevron-right": "chevron.right",
} as const satisfies Record<string, SFSymbol>;
 
function resolveSf(
  map: Record<string, SFSymbol>,
  name: string
): SFSymbol | null {
  const sf = map[name];
  if (!sf || IOS_UNSAFE_SF.has(sf)) {
    return null;
  }
  return sf;
}
 
/** Retourne null si pas de correspondance ou symbole non sûr → police MaterialCommunityIcons. */
export function mciToSf(name: string): SFSymbol | null {
  return resolveSf(MCI_TO_SF as Record<string, SFSymbol>, name);
}
 
export function ionToSf(name: string): SFSymbol | null {
  return resolveSf(ION_TO_SF as Record<string, SFSymbol>, name);
}
 
export function miToSf(name: string): SFSymbol | null {
  return resolveSf(MI_TO_SF as Record<string, SFSymbol>, name);
}
